#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	string s;
	cin >> s;

	int pipeCount = 0;

	int score = 0;
	bool lastOpen = false;
	for (auto c : s)
	{
		if (c == '(')
		{
			score++;
			lastOpen = true;
		}
		else
		{
			score--;
			pipeCount += lastOpen ? score : 1;
			lastOpen = false;
		}
	}

	cout << pipeCount;
}
