```cpp
#include <iostream>
#include <algorithm>
#include <stack>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int t;
	cin >> t;

	int balancedStringCount = 0;

	while (t--)
	{
		string s;
		cin >> s;

		stack<char> st;
		bool isBalanced = true;
		for (auto c : s)
		{
			if (c == '(')
				st.push(c);
			else
			{
				if (st.empty() || st.top() != '(')
				{
					isBalanced = false;
					break;
				}
				st.pop();
			}
		}

		if (st.empty() == false)
			isBalanced = false;

		cout << (isBalanced ? "YES" : "NO") << '\n';
	}
}
```
