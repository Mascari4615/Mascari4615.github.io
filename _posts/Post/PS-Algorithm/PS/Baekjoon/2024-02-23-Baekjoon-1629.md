## **💫 C++ 풀이**

---

```cpp
#include <iostream>

using namespace std;

using ll = long long;
ll a, b, c;

// (ll a, ll b, ll c)
// 재귀 동안 a와 c는 변하지 않는 값이니까
ll calc(ll nb)
{
	// (nb <= 1)
	// nb가 0이 되는 경우는 없다
	if (nb == 1)
		return a % c;

	ll temp = calc(nb / 2);
	temp = (temp * temp) % c;

	// nb % 2 == 1
	// 홀수 판별
	if (nb & 1)
		temp = (temp * a % c) % c;
	return temp;
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	cin >> a >> b >> c;
	cout << calc(b);
}
```
