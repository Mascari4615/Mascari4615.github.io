```cpp
#include <iostream>
#include <algorithm>
#include <queue>

using namespace std;

int board[502][502];
int dist[502][502];

int dx[4]{1, -1, 0, 0};
int dy[4]{0, 0, 1, -1};

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int n, m;
	cin >> n >> m;

	for (int i = 0; i < n; i++)
	{
		// MISS
		// í™œì„±í™”ëœ íƒ€ì¼ ìœ ë¬´ë¥¼ êµ³ì´ ìˆ«ìë¡œ íŒë‹¨í•˜ì§€ ì•Šì•„ë„ ë¼ì„œ
		// boardë¥¼ string[]ë¡œ ë§Œë“¤ì—ˆì–´ë„ ëìŒ
		string s;
		cin >> s;
		for (int j = 0; j < m; j++)
			board[i][j] = s[j] - '0';
	}

	for (int i = 0; i < n; i++)
		fill(dist[i], dist[i] + m, -1);

	queue<pair<int, int>> q;

	dist[0][0] = 0;
	q.push({0, 0});

	while (q.empty() == false)
	{
		pair<int, int> cur = q.front();
		q.pop();

		for (int k = 0; k < 4; k++)
		{
			int newX = cur.first + dx[k];
			int newY = cur.second + dy[k];

			if (newX < 0 || newX >= n || newY < 0 || newY >= m)
				continue;
			if ((dist[newX][newY] >= 0) || (board[newX][newY] == 0))
				continue;

			dist[newX][newY] = dist[cur.first][cur.second] + 1;
			q.push({newX, newY});
		}
	}

	cout << dist[n - 1][m - 1] + 1;
}
```
