## **💫 C++ 풀이**

---

```cpp
#include <iostream>
#include <algorithm>
#include <queue>

using namespace std;

int board[1005][1005];
int day[1005][1005];

int dx[4]{1, -1, 0, 0};
int dy[4]{0, 0, 1, -1};

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	// MISS
	// 가로 세로 반대로 입력 받았었음
	int n, m;
	cin >> m >> n;

	queue<pair<int, int>> q;

	for (int i = 0; i < n; i++)
		fill(day[i], day[i] + m, -1);

	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
		{
			cin >> board[i][j];

			if (board[i][j] == 1)
			{
				day[i][j] = 0;
				q.push({i, j});
			}
		}

	while (q.empty() == false)
	{
		pair<int, int> cur = q.front();
		q.pop();

		for (int k = 0; k < 4; k++)
		{
			int newX = cur.first + dx[k];
			int newY = cur.second + dy[k];

			if (newX < 0 || newX >= n || newY < 0 || newY >= m)
				continue;
			if ((day[newX][newY] > -1) || (board[newX][newY] == -1))
				continue;

			day[newX][newY] = day[cur.first][cur.second] + 1;
			q.push({newX, newY});
		}
	}

	int maxDay = 0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
		{
			if ((board[i][j] == 0) && (day[i][j] == -1))
			{
				cout << -1;
				return 0;
			}

			if (maxDay < day[i][j])
				maxDay = day[i][j];
		}

	cout << maxDay;
}
```
