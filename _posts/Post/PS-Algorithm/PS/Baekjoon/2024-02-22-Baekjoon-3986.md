## **💫 C++ 풀이**

---

```cpp
#include <iostream>
#include <algorithm>
#include <stack>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int n;
	cin >> n;

	int balancedStringCount = 0;

	while (n--)
	{
		string s;
		cin >> s;

		stack<char> st;
		for (auto c : s)
		{
			if (st.empty() == false && st.top() == c)
			{
				st.pop();
			}
			else
			{
				st.push(c);
			}
		}

		if (st.empty())
			balancedStringCount++;
	}

	cout << balancedStringCount;
}
```
