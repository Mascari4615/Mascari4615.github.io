## **π’« C++ ν’€μ΄**

---

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <cmath>

using namespace std;

int main()
{
	int n;
	cin >> n;

	if (n == 0)
	{
		cout << 0;
		return 0;
	}

	priority_queue<int> difficulties;

	for (int i = 0; i < n; i++)
	{
		int temp;
		cin >> temp;
		difficulties.push(temp);
	}

	// 30% μ μ‚¬ν‰κ· , μ„ μ•„λ 15% μ μ™Έ
	int cut = round(n * .15f);
	int sum = 0;

	// κ°€μ¥ μ‘μ€ 'cut'κ°μ μ«μ μ μ™Έ
	for (int i = 0; i < n - cut; i++)
	{
		// κ°€μ¥ ν° 'cut'κ°μ μ«μ μ μ™Έ
		if (i >= cut)
			sum += difficulties.top();
		difficulties.pop();
	}

	int result = round((float)sum / (n - cut * 2));
	cout << result;
}
```
