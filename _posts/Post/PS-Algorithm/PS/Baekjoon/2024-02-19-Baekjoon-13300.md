```cpp
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int n, k, s, y, minRoom = 0;
	cin >> n >> k;

	int numOfStudents[15]; // 0 ~ 5, 6 ~ 11
	fill(numOfStudents, numOfStudents + 15, 0);

	for (int i = 0; i < n; i++)
	{
		cin >> s >> y;
		numOfStudents[(s * 6) + y]++;
	}

	for (auto nos : numOfStudents)
		minRoom += (nos / k) + (nos % k != 0);

	cout << minRoom;
}
```
