## **💫 C++ 풀이**

---

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	string s1, s2;
	cin >> s1 >> s2;

	int diff[26];
	fill(diff, diff + 26, 0);

	for (auto c : s1)
		diff[c - 'a']++;
	for (auto c : s2)
		diff[c - 'a']--;

	int total = 0;
	for (auto d : diff)
	{
		if (d < 0)
			d = -d;

		total += d;
	}

	cout << total;
}
```
