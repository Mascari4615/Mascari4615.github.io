## **💫 C++ 풀이**

---

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int a, b, c;
	cin >> a >> b >> c;

	int mul = a * b * c;

	int count[10];
	fill(count, count + 10, 0);

	for (; mul != 0; mul /= 10)
		count[mul % 10]++;

	for (int i = 0; i < 10; i++)
		cout << count[i] << '\n';
}
```
