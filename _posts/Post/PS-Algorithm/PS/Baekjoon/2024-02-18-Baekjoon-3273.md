## **💫 C++ 풀이**

---

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

// 크기를 2000005로 잡으면, 아래 'A' if문이 없어도 됨 (대신 메모리를 더 씀)
bool arr[1000005];

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	int n, a, x;
	cin >> n;

	for (int i = 0; i < n; i++)
	{
		cin >> a;
		arr[a] = true;
	}

	cin >> x;

	int pairCount = 0;
	for (int i = 1; i < (x + 1) / 2; i++)
	{
		if (arr[i] == true)
			if (x - i < 1000000) // 'A'
				if (arr[x - i] == true)
					pairCount++;
	}

	cout << pairCount;
}
```
